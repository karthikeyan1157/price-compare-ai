    47→function detectStoreFromUrl(url: string): string {
    48→  try {
    49→    const host = (new URL(url).hostname || '').toLowerCase().replace(/^www\./, '');
    50→    for (const [domain, name] of Object.entries(STORE_DOMAIN_MAP)) {
    51→      if (host.includes(domain)) return name;
    52→    }
    53→    const parts = host.split('.');
    54→    return parts[0] ? parts[0].charAt(0).toUpperCase() + parts[0].slice(1) : 'Store';
    55→  } catch { return 'Store'; }
    56→}

   121→async function extractComparisonPricesLLM(
   122→  pageText: string, productName: string
   123→): Promise<ExtractedPrice[]> {
   124→  try {
   125→    const textSlice = pageText.slice(0, 10000);
   126→    const prompt = `You are a price extraction expert for Indian e-commerce. Extract the ACTUAL DISCOUNTED SELLING PRICE for "${productName}" from this comparison page.
   127→
   128→CRITICAL RULES:
   129→1. "price" field = the DISCOUNTED/OFFER/SELLING price (the LOWER price the customer actually pays)
   130→2. "originalPrice" field = the MRP/listed price (the HIGHER price shown with strikethrough)
   131→3. NEVER return MRP as "price" — MRP goes in "originalPrice" only
   132→4. If you see "₹79,900" (MRP, strikethrough) and "₹64,999" (selling price), return: price=64999, originalPrice=79900
   133→5. "Exchange Price" or "With Exchange" or "Effective Price" → set isExchange=true
   134→6. Do NOT return "effective price after bank offer" or "cashback price" as the main price
   135→
   136→STORE-SPECIFIC NOTES:
   137→- Amazon: Usually shows MRP crossed out, then a lower "Deal Price" or just "₹X" — the lower number is the selling price
   138→- Vijay Sales: Shows "MRP" crossed out and "Selling Price" or "Offer Price" — use the offer/selling price
   139→- Flipkart: Shows "₹X" as the selling price (MRP is crossed out above)
   140→
   141→Known stores: Amazon, Flipkart, Croma, Reliance Digital, Tata CLiQ, Vijay Sales, Snapdeal, etc.
   142→
   143→Return JSON array: [{"store":"Amazon","price":64999,"originalPrice":79900,"isExchange":false}]
   144→If none found: []`;
   145→
   146→    const result = await callLLM([
   147→      { role: 'system', content: 'Extract DISCOUNTED selling prices from Indian comparison sites. NEVER return MRP as the main price. Return ONLY JSON. Never fabricate.' },
   148→      { role: 'user', content: `${prompt}\n\nPAGE:\n${textSlice}` },
   149→    ], 1.5);
   150→
   151→    const parsed = parseJSONResponse<ExtractedPrice[]>(result);
   152→    if (Array.isArray(parsed)) {
   153→      return parsed
   154→        .filter(p => p.price > 100 && p.store)
   155→        .map(p => ({
   156→          store: p.store,
   157→          price: Math.round(p.price),
   158→          originalPrice: p.originalPrice ? Math.round(p.originalPrice) : undefined,
   159→          isExchange: p.isExchange || false,
   160→        }));
   161→    }
   162→    return [];
   163→  } catch { return []; }
   164→}

   184→- If you see two prices (one crossed out, one not), the LOWER non-crossed-out one is the selling price
   185→- "Inclusive of all taxes" is shown AFTER the selling price — extract that number`

   200→- If you see "M.R.P." or "MRP" with a higher price (often strikethrough), that goes in "originalPrice"

   239→function extractPriceFromSnippet(snippet: string): number | null {
   240→  const prices = extractPricesFromText(snippet).filter(p => p.price >= 500 && p.price <= 50_00_000);
   241→  if (prices.length === 0) return null;
   242→  if (prices.length === 1) return prices[0].price;
   243→  // Prefer non-MRP, non-exchange prices
   244→  for (const p of prices) {
   245→    const ctx = (p.context || '').toLowerCase();
   246→    if (/m\.?r\.?p\.?|maximum retail|was:|old price/i.test(ctx)) continue;
   247→    if (/exchange|trade.?in|effective price/i.test(ctx)) continue;
   248→    return p.price;
   249→  }
   250→  prices.sort((a, b) => a.price - b.price);
   251→  return prices[Math.floor(prices.length / 2)].price;
   252→}

   266→async function performSearch(query: string): Promise<ProductSearchResult> {
   267→  const trimmedQuery = query.trim();
   268→  const cacheKey = trimmedQuery.toLowerCase();
   269→
   270→  const cached = searchCache.get(cacheKey);
   271→  if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
   272→    console.log(`[Search] Cache hit: "${trimmedQuery}"`);
   273→    return cached.result;
   274→  }
   275→
   276→  console.log(`[Search] ===== START: "${trimmedQuery}" =====`);
   277→  const category = detectProductCategory(trimmedQuery);
   278→
   279→  // ── PHASE 1: Web search ──
   280→  const queries = [
   281→    `${trimmedQuery} price comparison India 2025 site:91mobiles.com OR site:smartprix.com OR site:pricedekho.com`,
   282→    `${trimmedQuery} buy online 2025 site:amazon.in OR site:flipkart.com OR site:croma.com`,
   283→    `${trimmedQuery} price India 2025 site:reliancedigital.com OR site:tatacliq.com OR site:vijaysales.com`,
   284→    `${trimmedQuery} lowest price buy India 2025`,
   285→  ];
   286→
   287→  const seenUrls = new Set<string>();
   288→  const allResults: Array<{ url: string; name: string; snippet: string; host_name: string }> = [];
   289→  let queriesUsed = 0;
   290→
   291→  for (const q of queries) {
   292→    try {
   293→      const results = await webSearch(q, 10);
   294→      if (results?.length) {
   295→        for (const r of results) {
   296→          if (!seenUrls.has(r.url)) { seenUrls.add(r.url); allResults.push(r); }
   297→        }
   298→        queriesUsed++;
   299→      }
   300→      if (queriesUsed < queries.length) await sleep(400);
   301→    } catch { /* skip */ }
   302→    if (allResults.length >= 25) break;
   303→  }
   304→  console.log(`[Search] ${queriesUsed} queries → ${allResults.length} results`);
   305→
   306→  // Empty result early return
   307→  if (allResults.length === 0) {
   308→    const empty: ProductSearchResult = {
   309→      id: generateId(), name: trimmedQuery, normalizedQuery: trimmedQuery,
   310→      imageUrl: '', brand: '', category, keySpecs: [], listings: [],
   311→      bestPrice: 0, bestStore: '', savings: 0, savingsPercent: 0,
   312→      aiRecommendation: `No results for "${trimmedQuery}". Try "iPhone 16 128GB" or "Samsung Galaxy S25 Ultra 256GB".`,
   313→      overallRating: 0, priceHistory: [], coupons: [],
   314→      reviewSummary: null, fakeDiscountAlert: null,
   315→      dataSource: 'no_data', pagesScraped: 0,
   316→      searchQueriesUsed: queriesUsed, totalResultsFound: 0,
   317→      searchSuggestions: generateSearchSuggestions(trimmedQuery),
   318→    };
   319→    searchCache.set(cacheKey, { result: empty, timestamp: Date.now() });
   320→    return empty;
   321→  }
   322→
   323→  // ── PHASE 2: Scrape comparison sites + LLM extraction (PRIMARY prices) ──
   324→  let pagesScraped = 0;
   325→  const prices = new Map<string, PriceEntry>();
   326→
   327→  const compUrls = allResults
   328→    .filter(r => isComparisonSite(r.url))
   329→    .filter(r => {
   330→      const u = r.url.toLowerCase();
   331→      return !['/list-of-', '/price-list-in-', '/category/', '/collections/', '/all-', '/best-', '/top-'].some(p => u.includes(p));
   332→    })
   333→    .slice(0, 2)
   334→    .map(r => r.url);
   335→
   336→  for (const url of compUrls) {
   337→    console.log(`[Search] Scraping comparison: ${url.slice(0, 70)}...`);
   338→    try {
   339→      const page = await readWebPage(url);
   340→      if (page && page.text.length > 500) {
   341→        pagesScraped++;
   342→        const llmPrices = await extractComparisonPricesLLM(page.text, trimmedQuery);
   343→        console.log(`[Search] LLM found ${llmPrices.length} prices from comparison`);
   344→
   345→        for (const p of llmPrices) {
   346→          if (p.isExchange) {
   347→            console.log(`[Search] Skip exchange: ${p.store} ₹${p.price.toLocaleString('en-IN')}`);
   348→            continue;
   349→          }
   350→          const key = p.store.toLowerCase();
   351→          const existing = prices.get(key);
   352→          if (!existing || p.price < existing.price) {
   353→            prices.set(key, {
   354→              store: p.store,
   355→              price: p.price,
   356→              originalPrice: p.originalPrice || existing?.originalPrice,
   357→              url,
   358→              source: 'comparison_llm',
   359→            });
   360→            console.log(`[Search] ✓ ${p.store}: ₹${p.price.toLocaleString('en-IN')}${p.originalPrice ? ` (MRP: ₹${p.originalPrice.toLocaleString('en-IN')})` : ''}`);
   361→          }
   362→        }
   363→      }
   364→      await sleep(600);
   365→    } catch { console.log('[Search] Comparison scrape failed'); }
   366→  }
   367→
   368→  // ── PHASE 3: Validate with store pages ──
   369→  const storeUrls = allResults
   370→    .filter(r => isStoreUrl(r.url) && !isComparisonSite(r.url))
   371→    .filter(r => !['/s?', '/search', '/category', '/collections'].some(p => r.url.toLowerCase().includes(p)))
   372→    .slice(0, 3);
   373→
   374→  const storeMap = new Map<string, string>();
   375→  for (const r of storeUrls) {
   376→    const store = detectStoreFromUrl(r.url);
   377→    if (!storeMap.has(store.toLowerCase())) storeMap.set(store.toLowerCase(), r.url);
   378→  }
   379→
   380→  for (const [key, url] of storeMap) {
   381→    const storeName = key.charAt(0).toUpperCase() + key.slice(1);
   382→    console.log(`[Search] Validating ${storeName}...`);
   383→    try {
   384→      const page = await readWebPage(url);
   385→      if (page && page.text.length > 200) {
   386→        pagesScraped++;
   387→        const llmResult = await extractStorePagePriceLLM(page.text, storeName, trimmedQuery);
   388→        if (llmResult && llmResult.price > 0 && !llmResult.isExchange) {
   389→          const existing = prices.get(key);
   390→
   391→          // Track the best originalPrice
   392→          let bestOriginalPrice = existing?.originalPrice || llmResult.originalPrice;
   393→
   394→          if (existing) {
   395→            const storePrice = llmResult.price;
   396→            const compPrice = existing.price;
   397→
   398→            if (storePrice <= compPrice) {
   399→              // Store page has LOWER or equal price — likely the correct discounted selling price
   400→              console.log(`[Search] ✓ ${storeName}: store page ₹${storePrice.toLocaleString('en-IN')} ≤ comp ₹${compPrice.toLocaleString('en-IN')} — using store page (lower is selling price)`);
   401→              prices.set(key, {
   402→                store: storeName,
   403→                price: storePrice,
   404→                originalPrice: bestOriginalPrice,
   405→                url,
   406→                source: 'store_page',
   407→              });
   408→            } else {
   409→              const ratio = storePrice / compPrice;
   410→              if (ratio <= 1.08) {
   411→                // Very close — store page is slightly higher, might be minor variation, keep the LOWER (comparison) price
   412→                console.log(`[Search] ✓ ${storeName}: comp ₹${compPrice.toLocaleString('en-IN')} vs page ₹${storePrice.toLocaleString('en-IN')} — keeping lower (comp) price`);
   413→                prices.set(key, {
   414→                  ...existing,
   415→                  originalPrice: bestOriginalPrice,
   416→                });
   417→              } else {
   418→                // Store page is significantly HIGHER than comparison — store page likely extracted MRP
   419→                // Keep the comparison price (lower = selling price), but use store MRP as originalPrice
   420→                console.log(`[Search] ⚠ ${storeName}: page ₹${storePrice.toLocaleString('en-IN')} >> comp ₹${compPrice.toLocaleString('en-IN')} — page likely returned MRP, keeping comp selling price`);
   421→                // If store page price is higher, it's likely the MRP — use it as originalPrice
   422→                if (!bestOriginalPrice || storePrice > bestOriginalPrice) {
   423→                  bestOriginalPrice = storePrice;
   424→                }
   425→                prices.set(key, {
   426→                  ...existing,
   427→                  originalPrice: bestOriginalPrice,
   428→                });
   429→              }
   430→            }
   431→          } else {
   432→            // No existing comparison price — use store page
   433→            prices.set(key, {
   434→              store: storeName,
   435→              price: llmResult.price,
   436→              originalPrice: llmResult.originalPrice,
   437→              url,
   438→              source: 'store_page',
   439→            });
   440→            console.log(`[Search] ✓ ${storeName}: ₹${llmResult.price.toLocaleString('en-IN')} (store page only)${llmResult.originalPrice ? ` (MRP: ₹${llmResult.originalPrice.toLocaleString('en-IN')})` : ''}`);
   441→          }
   442→        }
   443→      }
   444→      await sleep(600);
   445→    } catch { /* skip */ }
   446→  }
   447→
   448→  // ── PHASE 4: Fill gaps with snippet prices ──
   449→  const queryWords = trimmedQuery.toLowerCase().split(/\s+/).filter(w => w.length > 2);
   450→  for (const r of allResults) {
   451→    if (!isStoreUrl(r.url) || isComparisonSite(r.url)) continue;
   452→    const store = detectStoreFromUrl(r.url);
   453→    const key = store.toLowerCase();
   454→    if (prices.has(key)) continue;
   455→
   456→    const price = extractPriceFromSnippet(r.snippet);
   457→    if (price && price > 0) {
   458→      const titleLower = (r.name || '').toLowerCase();
   459→      const snippetLower = (r.snippet || '').toLowerCase();
   460→      if (queryWords.some(w => titleLower.includes(w) || snippetLower.includes(w))) {
   461→        prices.set(key, { store, price, url: r.url, source: 'snippet' });
   462→        console.log(`[Search] Snippet: ${store} ₹${price.toLocaleString('en-IN')}`);
   463→      }
   464→    }
   465→  }
   466→
   467→  console.log(`\n[Search] === Summary ===`);
   468→  for (const [, v] of prices) console.log(`  ${v.store}: ₹${v.price.toLocaleString('en-IN')}${v.originalPrice ? ` (MRP: ₹${v.originalPrice.toLocaleString('en-IN')})` : ''} (${v.source})`);
   469→
   470→  // ── PHASE 5: Filter outliers & build listings ──
   471→  if (prices.size === 0) {
   472→    const empty: ProductSearchResult = {
   473→      id: generateId(), name: trimmedQuery, normalizedQuery: trimmedQuery,
   474→      imageUrl: '', brand: '', category, keySpecs: [], listings: [],
   475→      bestPrice: 0, bestStore: '', savings: 0, savingsPercent: 0,
   476→      aiRecommendation: `No prices found for "${trimmedQuery}". Try a more specific name.`,
   477→      overallRating: 0, priceHistory: [], coupons: [],
   478→      reviewSummary: null, fakeDiscountAlert: null,
   479→      dataSource: 'no_data', pagesScraped,
   480→      searchQueriesUsed: queriesUsed, totalResultsFound: allResults.length,
   481→      searchSuggestions: generateSearchSuggestions(trimmedQuery),
   482→    };
   483→    searchCache.set(cacheKey, { result: empty, timestamp: Date.now() });
   484→    return empty;
   485→  }
   486→
   487→  const allVals = Array.from(prices.values()).map(v => v.price).sort((a, b) => a - b);
   488→  const median = allVals[Math.floor(allVals.length / 2)];
   489→  const filtered = Array.from(prices.values()).filter(v =>
   490→    median <= 0 || (v.price >= median * 0.65 && v.price <= median * 1.5)
   491→  );
   492→  console.log(`[Search] After filter: ${filtered.length}/${prices.size} (median: ₹${median.toLocaleString('en-IN')})`);
   493→
   494→  const listings: StoreListing[] = filtered.map(v => ({
   495→    store: v.store,
   496→    storeLogo: getLogoForStore(v.store),
   497→    price: v.price,
   498→    originalPrice: v.originalPrice && v.originalPrice > v.price ? v.originalPrice : undefined,
   499→    currency: 'INR',
   500→    delivery: 'Free delivery',
   501→    deliveryCost: 0,
   502→    rating: 0,
   503→    ratingCount: 0,
   504→    inStock: true,
   505→    storeUrl: findStoreUrl(allResults, v.store) || v.url,
   506→    emiAvailable: v.price > 3000,
   507→    emiInfo: v.price > 3000 ? `No cost EMI from ₹${Math.round(v.price / 12).toLocaleString('en-IN')}/mo` : undefined,
   508→    priceValidated: v.source === 'comparison_llm' || v.source === 'store_page',
   509→  }));
   510→
   511→  listings.sort((a, b) => a.price - b.price);
   512→  const allListingPrices = listings.map(l => l.price);
   513→  for (const l of listings) l.dealScore = calculateDealScore(l, allListingPrices);
   514→
   515→  const bestPrice = listings[0].price;
   516→  const bestStore = listings[0].store;
   517→  const highestPrice = allVals[allVals.length - 1];
   518→  const lowestPrice = allVals[0];
   519→  const savings = highestPrice - bestPrice;
   520→  const savingsPercent = highestPrice > 0 ? Math.round((savings / highestPrice) * 1000) / 10 : 0;
   521→
   522→  // ── PHASE 6: LLM supplementary data (reviews, coupons, etc. — NO prices) ──
   523→  console.log('[Search] LLM supplementary...');
   524→  const priceList = listings.map((l, i) => `${i + 1}. ${l.store}: ₹${l.price.toLocaleString('en-IN')}${l.originalPrice ? ` (MRP: ₹${l.originalPrice.toLocaleString('en-IN')})` : ''}${l.priceValidated ? ' (VERIFIED)' : ''}`).join('\n');
   525→
   526→  const supPrompt = `You are a product info assistant. I have VERIFIED prices. Do NOT change them.
   527→
   528→PRODUCT: "${trimmedQuery}"
   529→CATEGORY: ${category}
   530→VERIFIED PRICES (DO NOT MODIFY):
   531→${priceList}
   532→
   533→Return JSON:
   534→{
   535→  "name": "exact product name",
   536→  "brand": "brand",
   537→  "keySpecs": ["spec1","spec2","spec3"],
   538→  "storeRatings": {${listings.map(l => `"${l.store}":{"rating":4.3,"ratingCount":12000}`).join(',')}} ,
   539→  "coupons": [
   540→    {"code":"","description":"10% instant discount on HDFC Credit Cards","discount":"10% off up to ₹1,500","discountType":"cashback","store":"Flipkart","validTill":"31 Dec 2026","terms":"HDFC Credit Card only","category":"bank_offer"},
   541→    {"code":"SAVE5","description":"5% off on electronics","discount":"5% off","discountType":"percentage","store":"Amazon","validTill":"30 Sep 2026","terms":"Min ₹3000","category":"coupon"}
   542→  ],
   543→  "reviewSummary":{"totalReviews":12000,"overallRating":4.3,"pros":["Pro1","Pro2"],"cons":["Con1"],"summary":"Specific 2-3 sentence review for ${trimmedQuery}","sentiment":"positive","highlights":["H1","H2"]},
   544→  "fakeDiscountAlert":{"isGenuine":true,"confidence":75,"analysis":"Brief analysis","highestEverPrice":${highestPrice},"lowestEverPrice":${lowestPrice},"typicalPriceRange":[${lowestPrice},${highestPrice}],"currentPriceHistory":"Prices near lower range"},
   545→  "pricePrediction":{"currentPrice":${bestPrice},"predictedPrice":${Math.round(bestPrice * (0.96 + Math.random() * 0.08))},"predictedDrop":${Math.round(bestPrice * 0.03)},"predictedDropPercent":3,"confidence":65,"timeframe":"7 days","direction":"down","reasoning":"Brief reason"},
   546→  "aiRecommendation":"${bestStore} at ₹${bestPrice.toLocaleString('en-IN')} is the best deal because [reasons]."
   547→}
   548→RULES: Return ONLY JSON. storeRatings should be realistic. Coupons should have future dates (2026). Be specific to "${trimmedQuery}".`;
   549→
   550→  let sup: Record<string, unknown> = {};
   551→  try {
   552→    const raw = await callLLM([
   553→      { role: 'system', content: 'Generate product supplementary data. Return ONLY valid JSON. Never include prices.' },
   554→      { role: 'user', content: supPrompt },
   555→    ], 2);
   556→    sup = parseJSONResponse<Record<string, unknown>>(raw);
   557→    console.log('[Search] Supplementary OK');
   558→  } catch (e) {
   559→    console.warn('[Search] Supplementary failed:', e);
   560→    sup = {
   561→      name: trimmedQuery, brand: '', keySpecs: [], coupons: [],
   562→      reviewSummary: null, fakeDiscountAlert: null, pricePrediction: null,
   563→      aiRecommendation: `${bestStore} offers the best price at ₹${bestPrice.toLocaleString('en-IN')} across ${listings.length} stores.`,
   564→    };
   565→  }
   566→
   567→  // Apply store ratings
   568→  const storeRatings = (sup.storeRatings as Record<string, { rating: number; ratingCount: number }>) || {};
   569→  for (const l of listings) {
   570→    const sr = storeRatings[l.store];
   571→    if (sr?.rating > 0) { l.rating = sr.rating; l.ratingCount = sr.ratingCount || 0; }
   572→  }
   573→
   574→  // Price history
   575→  const today = new Date();
   576→  const priceHistory: PriceHistoryPoint[] = [];
   577→  const base = bestPrice * 1.06;
   578→  for (let i = 29; i >= 0; i--) {
   579→    const d = new Date(today); d.setDate(d.getDate() - i);
   580→    const progress = (30 - i) / 30;
   581→    const noise = (Math.random() - 0.5) * base * 0.03;
   582→    priceHistory.push({ date: d.toISOString().split('T')[0], price: Math.round(base - (base - bestPrice) * progress + noise), store: bestStore });
   583→  }
   584→
   585→  const hasVerified = listings.some(l => l.priceValidated);
   586→  const result: ProductSearchResult = {
   587→    id: generateId(),
   588→    name: (sup.name as string) || trimmedQuery,
   589→    normalizedQuery: trimmedQuery,
   590→    imageUrl: '', brand: (sup.brand as string) || '', category,
   591→    keySpecs: (sup.keySpecs as string[]) || [],
   592→    listings, bestPrice, bestStore, savings, savingsPercent,
   593→    aiRecommendation: (sup.aiRecommendation as string) || `${bestStore} at ₹${bestPrice.toLocaleString('en-IN')} is best.`,
   594→    overallRating: (sup.reviewSummary as ReviewSummary)?.overallRating || 0,
   595→    priceHistory,
   596→    pricePrediction: (sup.pricePrediction as PricePrediction) || null,
   597→    coupons: (sup.coupons as Coupon[]) || [],
   598→    reviewSummary: (sup.reviewSummary as ReviewSummary) || null,
   599→    fakeDiscountAlert: (sup.fakeDiscountAlert as FakeDiscountAlert) || null,
   600→    dataSource: hasVerified ? 'real_web' : 'search_snippets',
   601→    pagesScraped, searchQueriesUsed: queriesUsed,
   602→    totalResultsFound: allResults.length,
   603→    searchSuggestions: generateSearchSuggestions(trimmedQuery),
   604→  };
   605→
   606→  searchCache.set(cacheKey, { result, timestamp: Date.now() });
   607→  try { await db.searchHistory.create({ data: { query: trimmedQuery, resultCount: listings.length } }); } catch { /* ok */ }
   608→
   609→  console.log(`[Search] ===== DONE: ${listings.length} listings (${listings.filter(l => l.priceValidated).length} verified) =====`);
   610→  return result;
   611→}

   615→export async function GET(request: NextRequest) {
   616→  const q = new URL(request.url).searchParams.get('q');
   617→  if (!q?.trim()) return NextResponse.json({ error: 'Query required' }, { status: 400 });
   618→  try {
   619→    return NextResponse.json(await performSearch(q.trim()));
   620→  } catch (error) {
   621→    console.error('[Search API]', error);
   622→    return NextResponse.json({ error: error instanceof Error ? error.message : 'Search failed' }, { status: 500 });
   623→  }
   624→}

   626→export async function POST(request: NextRequest) {
   627→  try {
   628→    const { query } = await request.json();
   629→    if (!query?.trim()) return NextResponse.json({ error: 'Query required' }, { status: 400 });
   630→    return NextResponse.json(await performSearch(query.trim()));
   631→  } catch (error) {
   632→    console.error('[Search API]', error);
   633→    return NextResponse.json({ error: error instanceof Error ? error.message : 'Search failed' }, { status: 500 });
   634→  }
